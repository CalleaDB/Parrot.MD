    <script>

        function updateVisualization() {

            const hourlyRate = parseFloat(document.getElementById('hourlyRate').value);

            const hoursWorked = parseFloat(document.getElementById('hoursWorked').value);

            const connectionFeePercent = parseFloat(document.getElementById('connectionFee').value);

            const hasPodLeader = document.getElementById('podLeader').value === 'true';

            

            const grossCompensation = hourlyRate * hoursWorked;

            const revenueGenerated = grossCompensation * (connectionFeePercent / 100);

            

            // Company always gets 15%, plus Pod Leader's 2% if no Pod Leader present

            const companyShare = hasPodLeader ? 

                grossCompensation * 0.15 : 

                grossCompensation * 0.17;

            const podLeaderShare = hasPodLeader ? grossCompensation * 0.02 : 0;

            const bdmShare = revenueGenerated - companyShare - podLeaderShare;

            

            // Update breakdown

            updateBreakdown(companyShare, podLeaderShare, bdmShare, hasPodLeader);

            

            // Update compensation details

            updateCompensationDetails(grossCompensation, revenueGenerated, connectionFeePercent, bdmShare);

            

            // Update total BDM earnings

            document.getElementById('totalBdmEarnings').textContent = `$${bdmShare.toFixed(0)}`;

        }

        


        

        function updateBreakdown(companyShare, podLeaderShare, bdmShare, hasPodLeader) {

            const breakdown = document.getElementById('breakdown');

            const connectionFeePercent = parseFloat(document.getElementById('connectionFee').value);

            const bdmPercent = connectionFeePercent - 17;

            

            let html = `

                <div class="breakdown-item company">

                    <span>Company ($${hasPodLeader ? '15' : '17'}%)</span>

                    <span>$${companyShare.toFixed(0)}</span>

                </div>

            `;

            

            if (hasPodLeader) {

                html += `

                    <div class="breakdown-item pod-leader">

                        <span>Pod Leader (2%)</span>

                        <span>$${podLeaderShare.toFixed(0)}</span>

                    </div>

                `;

            }

            

            html += `

                <div class="breakdown-item bdm">

                    <span>BDM (${bdmPercent}%)</span>

                    <span>${bdmShare.toFixed(0)}</span>

                </div>

            `;

            

            breakdown.innerHTML = html;

        }

        

        function updateCompensationDetails(grossCompensation, revenueGenerated, connectionFeePercent, bdmShare) {

            const details = document.getElementById('compensationDetails');

            const bdmPercent = connectionFeePercent - 17;

            

            details.innerHTML = `

                <div style="padding: 20px; background: rgba(211, 199, 146, 0.2); border-radius: 12px; border: 1px solid #D3C792;">

                    <h4 style="color: #0D3049; margin-bottom: 15px; font-size: 1.1rem;">Shift Calculation</h4>

                    <div style="display: flex; justify-content: space-between; margin-bottom: 8px;">

                        <span>Gross Compensation:</span>

                        <strong>$${grossCompensation.toFixed(0)}</strong>

                    </div>

                    <div style="display: flex; justify-content: space-between; margin-bottom: 8px;">

                        <span>Connection Fee (${connectionFeePercent}%):</span>

                        <strong>$${revenueGenerated.toFixed(0)}</strong>

                    </div>

                    <div style="display: flex; justify-content: space-between; margin-bottom: 8px;">

                        <span>Your Share (${bdmPercent}%):</span>

                        <strong style="color: #F46300;">$${bdmShare.toFixed(0)}</strong>

                    </div>

                </div>

            `;

        }

        document.getElementById('podLeader').addEventListener('change', updateVisualization);

        

        // Initial update

        updateVisualization();
